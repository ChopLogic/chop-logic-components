# ChopLogicTabs

Versatile tab component that allows you to switch between different pieces of content by clicking on corresponding tabs.

### Example

<Canvas>
  <Story of={ChopLogicTabsStories.Default} />
</Canvas>

### Usage

```tsx
import React from 'react';
import ChopLogicTabs, { ChopLogicTabItem } from 'your-library/ChopLogicTabs';

const tabs: ChopLogicTabItem[] = [
  { id: 'tab1', title: 'Tab 1', tabContent: <div>Content for Tab 1</div> },
  { id: 'tab2', title: 'Tab 2', tabContent: <div>Content for Tab 2</div> },
  { id: 'tab3', title: 'Tab 3', tabContent: <div>Content for Tab 3</div> },
];

const ExampleComponent = () => <ChopLogicTabs tabs={tabs} defaultTabId='tab1' />;

export default ExampleComponent;
```

### Props

- **tabs**
  - Type: `ChopLogicTabItem[]`
  - Description: An array of tab items, each containing an id, title, and the content to display.
- **defaultTabId**
  - Type: `string`
  - Description: The id of the tab that should be active by default.
- **mode**
  - Type: `'vertical' | 'horizontal'`
  - Description: The orientation of the tabs. Defaults to `'horizontal'`.

### Types

```ts
type ChopLogicTabsProps = React.HTMLAttributes<HTMLDivElement> & {
  tabs: ChopLogicTabItem[];
  defaultTabId?: string;
  mode?: ChopLogicTabsMode;
};
```

```ts
type ChopLogicTabItem = {
  content: React.ReactElement;
  title: string;
  id: string;
  disabled?: boolean;
};
```

```ts
type ChopLogicTabsMode = 'vertical' | 'horizontal';
```
