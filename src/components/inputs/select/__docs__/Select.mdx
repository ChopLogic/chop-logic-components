# Select

Select component with custom styling and behavior.

#### Example

<Canvas of={Select.Default} />

## Usage

```tsx
import { ChopLogicSelect } from 'sld-ui';

const LANGUAGES: SelectValue[] = [
  {
    id: '57b5a856-bc4b-4c2f-8295-9f44434fafc7',
    label: 'English',
  },
  {
    id: 'b4b08a09-e287-400b-ba65-afd6ae1f6fd7',
    label: 'Russian',
  },
  {
    id: '641fc722-af5a-4426-8f8a-1566cc492b91',
    label: 'German',
  },
];

const Example = () => {
  return <ChopLogicSelect id='example-select' name='laguage' values={LANGUAGES} label='Select your language' />;
};

export default Example;
```
