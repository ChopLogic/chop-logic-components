# useModalFocusTrap

The `useModalFocusTrap` hook ensures that focus is trapped within an open modal window, preventing the focus from moving outside of the modal.

### Usage

```tsx
import React, { useRef, useState } from 'react';
import { useModalFocusTrap } from 'chop-logic-components';

const ExampleModal = () => {
  const modalRef = useRef<HTMLDivElement>(null);
  const [isOpened, setIsOpened] = useState(false);

  useModalFocusTrap({ modalRef, isOpened });

  return (
    <div>
      <button onClick={() => setIsOpened(true)}>Open Modal</button>
      {isOpened && (
        <div ref={modalRef} tabIndex={-1}>
          <button onClick={() => setIsOpened(false)}>Close Modal</button>
          <input type='text' placeholder='Focusable Input' />
        </div>
      )}
    </div>
  );
};

export default ExampleModal;
```

### Params

- **modalRef**
  - Type: `React.RefObject<HTMLDivElement>`
  - Description: A ref to the modal element that will be used to trap the focus within.
- **isOpened**
  - Type: `boolean`
  - Description: A boolean indicating whether the modal is opened or not.

### Returns

- `void`
